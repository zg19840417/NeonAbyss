> **版本**：v2.0
> **版本日期**：2026-04-06
> **用途**：定义每日任务系统的规则、任务列表和奖励

---

## 一、系统概述

### 1.1 定义

每日任务是每天刷新的临时目标，完成后可领取奖励。每日任务引导玩家进行日常游戏行为，保持日活。

### 1.2 入口

主界面 → 底部导航"任务" → "每日任务"标签页

### 1.3 核心规则

- 每日任务**每天凌晨5:00刷新**（服务器时间）
- 刷新后未领取的奖励**自动消失**
- 每日任务分为**普通任务**和**精英任务**两个层级
- 全部普通任务完成后可领取**每日活跃度宝箱**
- 精英任务为额外挑战，完成后获得额外奖励

## 五、与配置表的关联

### 5.1 quest.xlsx 表结构

| 字段名 | 类型 | 说明 |
|--------|------|------|
| questId | string | 任务唯一ID |
| questType | string | 任务类型（daily_normal/daily_elite/weekly） |
| name | string | 任务名称 |
| description | string | 任务描述 |
| condition | string | 达成条件标识 |
| targetCount | int | 目标次数 |
| rewards | json | 奖励列表 |
| activityPoints | int | 完成后获得的活跃度（0=不获得） |
| weight | int | 抽取权重（权重越高越容易被抽到） |
| isExclusive | bool | 是否可与同类任务同时出现 |

### 5.2 数值策划配表要点

1. 每个任务一条记录，`questType`区分每日普通/每日精英/每周
2. `weight`字段控制任务被抽中的概率
3. `rewards`为JSON数组，支持多种奖励类型
4. 活跃度宝箱单独配置，不放在任务奖励中
5. 每周任务的刷新时间与每日任务相同（周一凌晨5:00）

## 💻 程序实现指南

> 本节提取本文档中与程序代码实现相关的所有内容，以可直接执行的方式呈现。

### 1. 每日任务刷新逻辑

```javascript
class DailyQuestSystem {
  constructor() {
    this.quests = []; // 当日活跃任务列表
    this.activityPoints = 0; // 当日活跃度
    this.claimedChests = []; // 已领取的宝箱ID列表
  }

  // 每日5:00刷新
  checkAndRefresh() {
    const now = new Date();
    const lastRefresh = new Date(playerData.dailyQuest.lastRefreshTime);
    
    // 判断是否跨天（5:00为分界线）
    const isNewDay = this.isNewDay(lastRefresh, now, 5);
    
    if (isNewDay) {
      this.refreshDailyQuests();
    }
  }

  isNewDay(lastTime, currentTime, hour) {
    const lastDate = new Date(lastTime);
    lastDate.setHours(hour, 0, 0, 0);
    const currentDate = new Date(currentTime);
    currentDate.setHours(hour, 0, 0, 0);
    return currentDate > lastDate;
  }

  refreshDailyQuests() {
    // 重置活跃度
    this.activityPoints = 0;
    this.claimedChests = [];

    // 从普通任务池中随机抽取4个
    const normalPool = QuestConfig.getByType('daily_normal');
    const selectedNormal = this.randomSelect(normalPool, 4, {
      exclude: playerData.dailyQuest.yesterdayNormalIds,
      minChange: 1, // 至少替换1个
    });

    // 从精英任务池中随机抽取2个
    const elitePool = QuestConfig.getByType('daily_elite');
    const selectedElite = this.randomSelect(elitePool, 2, {
      exclude: [],
    });

    // 初始化任务状态
    this.quests = [
      ...selectedNormal.map(q => ({ ...q, progress: 0, completed: false, rewarded: false })),
      ...selectedElite.map(q => ({ ...q, progress: 0, completed: false, rewarded: false })),
    ];

    // 记录昨日任务（用于去重）
    playerData.dailyQuest.yesterdayNormalIds = selectedNormal.map(q => q.questId);
    playerData.dailyQuest.lastRefreshTime = Date.now();
    this.save();
  }

  // 带权重的随机抽取
  randomSelect(pool, count, options = {}) {
    const available = pool.filter(q => !options.exclude?.includes(q.questId));
    
    // 如果需要至少替换1个
    if (options.minChange && options.exclude) {
      const changed = available.filter(q => !options.exclude.includes(q.questId));
      if (changed.length >= options.minChange) {
        // 至少选minChange个新任务
        const newOnes = this.weightedRandom(changed, options.minChange);
        const rest = this.weightedRandom(
          available.filter(q => !newOnes.includes(q)),
          count - options.minChange
        );
        return [...newOnes, ...rest];
      }
    }

    return this.weightedRandom(available, count);
  }

  weightedRandom(pool, count) {
    const totalWeight = pool.reduce((sum, q) => sum + q.weight, 0);
    const selected = [];
    const remaining = [...pool];

    for (let i = 0; i < count && remaining.length > 0; i++) {
      const currentWeight = remaining.reduce((sum, q) => sum + q.weight, 0);
      let random = Math.random() * currentWeight;
      
      for (let j = 0; j < remaining.length; j++) {
        random -= remaining[j].weight;
        if (random <= 0) {
          selected.push(remaining.splice(j, 1)[0]);
          break;
        }
      }
    }

    return selected;
  }
}
```

### 2. 任务进度追踪逻辑

// 事件触发时更新相关任务进度
function onGameEvent(eventName, data) {
  const questSystem = DailyQuestSystem.getInstance();

  for (const quest of questSystem.quests) {
    if (quest.completed) continue;

    // 匹配任务条件与事件
    if (matchCondition(quest.condition, eventName)) {
      quest.progress += getProgressIncrement(quest.condition, data);
      
      // 检查是否完成
      if (quest.progress >= quest.targetCount) {
        quest.progress = quest.targetCount;
        quest.completed = true;
        
        // 增加活跃度
        questSystem.activityPoints += quest.activityPoints;
        
        EventBus.emit('quest:completed', { questId: quest.questId });
      }

      EventBus.emit('quest:progress_update', {
        questId: quest.questId,
        progress: quest.progress,
        targetCount: quest.targetCount,
      });
    }
  }

  questSystem.save();
}

function claimQuestReward(questId) {

function claimActivityChest(chestId) {
  const chestConfig = ActivityChestConfig.get(chestId);
  
  // 检查活跃度是否足够
  if (questSystem.activityPoints < chestConfig.requiredPoints) return false;
  
  // 检查是否已领取
  if (questSystem.claimedChests.includes(chestId)) return false;

  // 发放宝箱奖励
  for (const reward of chestConfig.rewards) {
    RewardManager.grant(reward);
  }

  questSystem.claimedChests.push(chestId);
  EventBus.emit('chest:claimed', { chestId });
  questSystem.save();
  return true;
}

### 5. 每周任务独立刷新逻辑

```javascript
class WeeklyQuestSystem {
  constructor() {
    this.quests = [];
    this.claimedChests = [];
  }

  checkAndRefresh() {
    const now = new Date();
    const lastRefresh = new Date(playerData.weeklyQuest.lastRefreshTime);

    // 判断是否跨周（周一5:00为分界线）
    const isNewWeek = this.isNewWeek(lastRefresh, now);

    if (isNewWeek) {
      this.refreshWeeklyQuests();
    }
  }

  isNewWeek(lastTime, currentTime) {
    // 获取两个时间的周一日期
    const getLastMonday = (date) => {
      const d = new Date(date);
      const day = d.getDay();
      const diff = d.getDate() - day + (day === 0 ? -6 : 1);
      d.setDate(diff);
      d.setHours(5, 0, 0, 0);
      return d;
    };

    const lastMonday = getLastMonday(lastTime);
    const currentMonday = getLastMonday(currentTime);
    return currentMonday > lastMonday;
  }

  refreshWeeklyQuests() {
    // 每周任务全部显示，不需要随机抽取
    const allWeekly = QuestConfig.getByType('weekly');
    this.quests = allWeekly.map(q => ({
      ...q,
      progress: 0,
      completed: false,
      rewarded: false,
    }));
    this.claimedChests = [];
    playerData.weeklyQuest.lastRefreshTime = Date.now();
    this.save();
  }

  // 每周任务进度追踪（与每日任务共享事件监听）
  onGameEvent(eventName, data) {
    for (const quest of this.quests) {
      if (quest.completed) continue;
      if (matchCondition(quest.condition, eventName)) {
        quest.progress += getProgressIncrement(quest.condition, data);
        if (quest.progress >= quest.targetCount) {
          quest.progress = quest.targetCount;
          quest.completed = true;
          EventBus.emit('weekly_quest:completed', { questId: quest.questId });
        }
      }
    }
    this.save();
  }
}
```

### 6. EventBus 事件定义

| 事件名 | 触发时机 | 数据 |

| `quest:completed` | 每日任务完成 | `{ questId }` |
| `quest:reward_claimed` | 领取每日任务奖励 | `{ questId }` |
| `quest:progress_update` | 每日任务进度更新 | `{ questId, progress, targetCount }` |
| `weekly_quest:completed` | 每周任务完成 | `{ questId }` |
| `chest:claimed` | 领取活跃度宝箱 | `{ chestId }` |
| `activity:update` | 活跃度变动 | `{ points, maxPoints }` |
| `red_dot:update` | 红点状态更新 | `{ target, visible }` |
