## 一、系统概述

### 1.1 设计目的

图鉴系统是《废土元年》的收集与百科模块。玩家在探索废土的过程中，每遇到新的融合姬、敌人或物品，其信息将被自动记录到图鉴中。图鉴不仅提供世界观和角色背景故事的沉浸式阅读体验，还通过收集奖励机制驱动玩家主动探索和收集。

### 1.2 设计原则

| 原则 | 说明 |
|------|------|
| **渐进解锁** | 图鉴内容随游戏进程逐步解锁，未获得的条目显示为剪影 |
| **收集驱动** | 通过阶段性收集奖励激励玩家持续探索 |
| **沉浸叙事** | 每张图鉴卡片包含完整的背景故事，丰富世界观 |
| **便捷查阅** | 多维度筛选和搜索，快速定位目标条目 |
| **视觉反馈** | 收集进度可视化，解锁新条目时有明显反馈 |

### 1.3 系统入口

| 入口位置 | 打开方式 | 说明 |
|---------|---------|------|
| 主界面 → 庇护所 → 图鉴 | 点击庇护所界面中的图鉴图标 | 主入口 |
| 获得新卡牌时 | 结算界面点击卡牌 | 查看图鉴详情 |
| 探索遭遇新敌人 | 战斗结算自动提示 | 自动解锁 |
| 背包 → 物品详情 | 点击物品详情中的「查看图鉴」 | 跳转图鉴 |

### 1.4 图鉴总览数据

| 图鉴类别 | 总数量 | 首批实装 | 后续更新 |
|---------|--------|---------|---------|
| 融合姬图鉴 | 100张 | 60张 | 每版本+10张 |
| 敌人图鉴 | 50个 | 30个 | 每版本+5个 |
| 物品图鉴 | 200个 | 120个 | 随新内容同步 |
| **合计** | **350个** | **210个** | - |

### 2.2 图鉴解锁规则

| 图鉴类别 | 解锁条件 | 解锁时机 |

| 融合姬图鉴 | 获得对应融合姬卡牌 | 获得卡牌时自动解锁 |

| 物品图鉴 - 消耗品 | 首次获得对应物品 | 获得物品时自动解锁 |
| 物品图鉴 - 材料 | 首次获得对应材料 | 获得材料时自动解锁 |

## 三、融合姬图鉴

### 3.1 图鉴卡片信息

每张融合姬图鉴卡片包含以下信息：

| 信息项 | 说明 | 是否需要解锁 |
|--------|------|------------|
| 编号 | 图鉴编号（001-100） | 否 |
| 名称 | 融合姬名称 | 否（显示为"???"） |
| 品质 | N/R/SR/SSR/UR/LE | 否 |
| 种族 | 植物系/动物系/机械系/能量系/混合系 | 否 |
| 元素 | 水/火/风/光/暗 | 否 |
| 职业 | 坦克/输出/辅助 | 否 |
| 立绘 | 融合姬全身立绘 | 是 |
| 基础属性 | HP/攻击/防御/速度 | 是 |
| 技能列表 | 主动技能+被动技能 | 是 |
| 背景故事 | 200-500字角色背景 | 是 |
| 语音台词 | 3-5条角色语音 | 是 |
| 获取途径 | 召唤/关卡/活动 | 否 |

### 3.2 未解锁状态显示

- **立绘区域**：显示黑色剪影轮廓，品质颜色边框保留
- **名称**：显示为「???」
- **属性/技能/故事**：显示为「未解锁」灰色文字
- **提示文字**：「获得该融合姬后解锁详细信息」

// CodexManager.cs
public class CodexManager : MonoBehaviour
{
    public static CodexManager Instance { get; private set; }

    // 已解锁图鉴集合: codexType -> HashSet<codexId>
    private Dictionary<int, HashSet<string>> unlockedCodex = new();

    // 已领取的收集奖励: rewardId -> bool
    private Dictionary<int, bool> claimedRewards = new();

    /// <summary>
    /// 初始化图鉴系统
    /// </summary>
    public void Initialize()
    {
        unlockedCodex[1] = new HashSet<string>(); // 融合姬
        unlockedCodex[2] = new HashSet<string>(); // 敌人
        unlockedCodex[3] = new HashSet<string>(); // 物品

        // 从服务端加载已解锁数据
        StartCoroutine(LoadFromServer());
    }

    /// <summary>
    /// 解锁图鉴条目
    /// </summary>
    public void UnlockCodex(string codexId)
    {
        var config = CodexConfigTable.Get(codexId);
        if (config == null) return;

        int codexType = config.codex_type;

        // 检查是否已解锁
        if (unlockedCodex[codexType].Contains(codexId)) return;

        // 添加到已解锁集合
        unlockedCodex[codexType].Add(codexId);

        // 显示解锁提示
        CodexUI.ShowUnlockPopup(codexId);

        // 检查收集奖励
        CheckCollectionRewards(codexType);

        // 发送埋点
        AnalyticsManager.Track("codex_unlock", new { codexId, codexType });

        // 同步服务端
        StartCoroutine(SyncToServer(codexId));
    }

    /// <summary>
    /// 检查并触发收集奖励
    /// </summary>
    private void CheckCollectionRewards(int codexType)
    {
        int currentCount = unlockedCodex[codexType].Count;
        var rewards = CollectionRewardConfigTable.GetByType(codexType);

        foreach (var reward in rewards)
        {
            if (currentCount >= reward.required_count && !claimedRewards.ContainsKey(reward.reward_id))
            {
                // 显示可领取提示
                CodexUI.ShowRewardAvailable(reward);
            }
        }
    }

    /// <summary>
    /// 领取收集奖励
    /// </summary>
    public bool ClaimReward(int rewardId)
    {
        if (claimedRewards.ContainsKey(rewardId)) return false;

        var reward = CollectionRewardConfigTable.Get(rewardId);
        if (reward == null) return false;

        // 检查收集数量是否满足
        int currentCount = unlockedCodex[reward.codex_type].Count;
        if (currentCount < reward.required_count) return false;

        // 发放奖励
        foreach (var item in reward.reward_list)
        {
            RewardManager.GiveReward(item);
        }

        // 标记已领取
        claimedRewards[rewardId] = true;

        // 发送埋点
        AnalyticsManager.Track("codex_collection_claim", new { rewardId });

        return true;
    }

    /// <summary>
    /// 获取收集进度
    /// </summary>
    public CollectionProgress GetCollectionProgress(int codexType)
    {
        int current = unlockedCodex[codexType].Count;
        int total = CodexConfigTable.GetTotalCount(codexType);
        var nextReward = CollectionRewardConfigTable.GetNextReward(codexType, current);

        return new CollectionProgress
        {
            currentCount = current,
            totalCount = total,
            percentage = (float)current / total * 100f,
            nextReward = nextReward
        };
    }

    /// <summary>
    /// 获取图鉴详情
    /// </summary>
    public CodexDetail GetCodexDetail(string codexId)
    {
        var config = CodexConfigTable.Get(codexId);
        bool isUnlocked = unlockedCodex[config.codex_type].Contains(codexId);

        return new CodexDetail
        {
            config = config,
            isUnlocked = isUnlocked,
            // 未解锁时返回空数据
            storyText = isUnlocked ? config.story_text : "未解锁",
            skills = isUnlocked ? GetSkills(codexId) : new List<SkillData>(),
            stats = isUnlocked ? GetStats(codexId) : null
        };
    }
}

public class CodexGridPanel : MonoBehaviour
{
    [SerializeField] private Transform gridContainer;
    [SerializeField] private GameObject cardCellPrefab;
    [SerializeField] private CodexFilterBar filterBar;

    private List<CodexCardCell> cells = new();
    private int codexType;
    private CodexFilter currentFilter;

    /// <summary>
    /// 初始化图鉴网格
    /// </summary>
    public void Initialize(int type)
    {
        codexType = type;
        currentFilter = new CodexFilter();

        // 获取所有图鉴配置
        var configs = CodexConfigTable.GetByType(type);

        // 创建格子（使用对象池）
        cells.Clear();
        foreach (var config in configs)
        {
            var cell = ObjectPool.Instance.Get(cardCellPrefab, gridContainer)
                .GetComponent<CodexCardCell>();
            cell.Refresh(config, CodexManager.Instance.IsUnlocked(config.codex_id));
            cell.OnClick += () => ShowDetail(config.codex_id);
            cells.Add(cell);
        }

        // 注册筛选回调
        filterBar.OnFilterChanged += ApplyFilter;
    }

    /// <summary>
    /// 应用筛选
    /// </summary>
    private void ApplyFilter(CodexFilter filter)
    {
        currentFilter = filter;
        foreach (var cell in cells)
        {
            bool visible = cell.MatchesFilter(filter);
            cell.gameObject.SetActive(visible);
        }

        // 重新排列可见格子
        LayoutRebuilder.ForceRebuildLayoutImmediate(gridContainer as RectTransform);
    }

    /// <summary>
    /// 显示图鉴详情
    /// </summary>
    private void ShowDetail(string codexId)
    {
        var detail = CodexManager.Instance.GetCodexDetail(codexId);
        CodexUI.ShowDetailPanel(detail);
    }
}

    /// 检查是否匹配筛选条件
    /// </summary>

    {
        if (filter.quality.HasValue && config.quality != filter.quality.Value)
            return false;
        if (filter.element.HasValue && config.element != filter.element.Value)
            return false;
        if (filter.ownedOnly && !isUnlocked)
            return false;
        return true;
    }

    public void OnPointerClick(PointerEventData eventData)
    {
        OnClick?.Invoke();
    }

public class CollectionProgressPanel : MonoBehaviour

    [SerializeField] private ProgressBar[] progressBars; // 3个进度条

    /// 刷新收集进度
    /// </summary>

    {
        int[] types = { 1, 2, 3 }; // 融合姬/敌人/物品
        string[] names = { "融合姬", "敌人", "物品" };

        for (int i = 0; i < types.Length; i++)
        {
            var progress = CodexManager.Instance.GetCollectionProgress(types[i]);
            progressBars[i].UpdateProgress(progress.percentage / 100f);
            countTexts[i].text = $"{progress.currentCount}/{progress.totalCount}";

            if (progress.nextReward != null)
            {
                nextRewards[i].SetReward(progress.nextReward);
                nextRewards[i].SetProgress(
                    progress.currentCount,
                    progress.nextReward.required_count);
            }
            else
            {
                nextRewards[i].SetCompleted();
            }
        }

        // 刷新已获得称号
        RefreshTitles();
    }

    /// <summary>
    /// 刷新已获得称号列表
    /// </summary>
    private void RefreshTitles()
    {
        foreach (Transform child in titleContainer)
        {
            Destroy(child.gameObject);
        }

        var allRewards = CollectionRewardConfigTable.GetAll();
        foreach (var reward in allRewards)
        {
            if (CodexManager.Instance.IsRewardClaimed(reward.reward_id))
            {
                var titleObj = Instantiate(titlePrefab, titleContainer);
                titleObj.GetComponent<TitleTag>().Setup(
                    reward.title_name,
                    QualityColors.GetColor(reward.title_quality));
            }
        }
    }

    /// 播放品质对应解锁特效
    /// </summary>

        {
            0 => "Effects/Codex/unlock_n",    // N: 白色光点
            1 => "Effects/Codex/unlock_r",    // R: 绿色光点
            2 => "Effects/Codex/unlock_sr",   // SR: 蓝色光柱
            3 => "Effects/Codex/unlock_ssr",  // SSR: 紫色光柱+震动
            4 => "Effects/Codex/unlock_ur",   // UR: 橙色全屏
            5 => "Effects/Codex/unlock_le",   // LE: 金色全屏+彩虹
            _ => "Effects/Codex/unlock_n"
        };

        var effectPrefab = Resources.Load<ParticleSystem>(effectName);
        var effect = Instantiate(effectPrefab, transform);
        effect.Play();

        // SSR及以上添加屏幕震动
        if (quality >= 3)
        {
            CameraShaker.Instance.Shake(quality >= 4 ? 0.5f : 0.3f);
        }

        // 自动销毁特效
        Destroy(effect.gameObject, 2f);
    }

// CodexData.cs
public static class CodexData
{
    private const string SAVE_KEY = "codex_progress";

    [Serializable]
    public class SaveData
    {
        public string[] unlocked_fusion_girls;
        public string[] unlocked_enemies;
        public string[] unlocked_items;

    }

    /// 保存图鉴进度到本地
    /// </summary>
    public static void SaveLocal(SaveData data)
    {
        var json = JsonUtility.ToJson(data);
        PlayerPrefs.SetString(SAVE_KEY, json);
        PlayerPrefs.Save();
    }

    /// <summary>
    /// 从本地加载图鉴进度
    /// </summary>
    public static SaveData LoadLocal()
    {
        if (PlayerPrefs.HasKey(SAVE_KEY))
        {
            return JsonUtility.FromJson<SaveData>(PlayerPrefs.GetString(SAVE_KEY));
        }
        return new SaveData
        {
            unlocked_fusion_girls = new string[0],
            unlocked_enemies = new string[0],
            unlocked_items = new string[0],
            claimed_rewards = new int[0]
        };
    }

    /// <summary>
    /// 同步到服务端
    /// </summary>
    public static async Task SyncToServer(SaveData data)
    {
        await NetworkManager.PostAsync("codex/sync", data);
    }

    /// <summary>
    /// 从服务端拉取
    /// </summary>
    public static async Task<SaveData> PullFromServer()
    {
        var response = await NetworkManager.GetAsync("codex/pull");
        return JsonUtility.FromJson<SaveData>(response.data.ToString());
    }

### 10.8 关键注意事项

1. **性能优化**：100张融合姬立绘使用Addressable按需加载，避免一次性加载所有资源
2. **内存管理**：图鉴浏览时使用LRU缓存策略，最多缓存20张立绘，超出时释放最久未访问的
3. **离线支持**：已解锁图鉴的文本信息本地缓存，离线可查看；立绘离线时显示低分辨率缩略图
4. **预加载策略**：进入图鉴界面时预加载当前页+前后各1页的立绘资源
5. **数据安全**：图鉴解锁状态以服务端为准，客户端仅做展示缓存
6. **版本兼容**：新增图鉴条目时，旧版本客户端需能正常显示为"未解锁"状态
7. **搜索功能**：支持中文/拼音/编号搜索，搜索响应时间不超过200ms
8. **动画优化**：网格滚动时暂停所有解锁特效动画，滚动停止后恢复
9. **测试覆盖**：需覆盖首次解锁/重复解锁/离线解锁/版本更新等场景的自动化测试
