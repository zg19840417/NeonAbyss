## 混合系概述

混合系融合姬的身体同时融合了两种或三种不同种族的特征，是融合技术最复杂也最不稳定的产物。子种族包括机植混合（机械+植物）、动植混合（动物+植物）、机动混合（机械+动物）、能量混合（能量+其他）以及极其罕见的三系混合和混沌。混合系融合姬拥有多种种族的能力，但不同种族的基因在她体内不断冲突，需要付出巨大的代价来维持身体平衡。

当"灰烬联盟"发现她时，她正蜷缩在一个破裂的培养舱中。她的左臂覆盖着金属装甲板，手指是精密的机械关节；右臂则被藤蔓和苔藓覆盖，指甲是坚硬的木质结构。她的脊椎两侧生长着半透明的昆虫翅膀，但翅膀的骨架是碳纤维材质。她的头发是银白色的，其中混杂着光纤和细小的金属丝。她的左眼是金色的机械义眼，右眼则是翠绿色的植物复眼。她的皮肤在金属和植物覆盖的区域之间是正常的，但能看到微小的能量纹路在皮肤下流动。

唐锦瑟拥有三种种族的能力：机械臂可以变形为各种工具，植物臂能释放毒素和治愈孢子，昆虫翅膀赋予她短距离飞行能力。但三种种族的基因在她体内不断冲突，她经常经历剧烈的身体排斥反应——金属部分会生锈，植物部分会枯萎，动物部分会失控。她需要定期接受三种不同的维护治疗来维持身体平衡。尽管如此，她保持着令人惊讶的乐观态度，她说自己虽然是由碎片拼成的，但每一块碎片都是活着的。

### 3. 卡牌升级逻辑

```typescript
interface UpgradeCost {
  level: number;            // 目标等级
  myceliumCost: number;     // 菌丝消耗
  materialCost: Array<{     // 材料消耗
    itemId: string;
    count: number;
  }>;
}

// 品质系数映射
const QUALITY_COEFFICIENT: Record<string, number> = {
  'N': 1.0, 'R': 1.2, 'SR': 1.5, 'SSR': 2.0, 'UR': 2.5, 'LE': 3.0
};

// 升级消耗公式：cost = baseCost × level × qualityCoefficient
function getUpgradeCost(quality: string, level: number): number {
  const baseCost = 100; // 基础消耗（可配置）
  return Math.floor(baseCost * level * QUALITY_COEFFICIENT[quality]);
}

function upgradeCard(playerId: string, cardId: string): boolean {
  const card = getPlayerCard(playerId, cardId);
  const nextLevel = card.level + 1;

  // 等级上限 = 玩家声望等级
  const playerReputationLevel = getPlayerReputationLevel(playerId);
  if (nextLevel > playerReputationLevel) return false;

  const myceliumCost = getUpgradeCost(card.quality, nextLevel);

  // 检查菌丝是否足够
  if (!hasEnoughMycelium(playerId, myceliumCost)) return false;

  // 扣除菌丝
  deductMycelium(playerId, myceliumCost);

  // 提升属性
  const growthRates = getGrowthRates(card.quality, card.profession);
  card.hp += growthRates.hp;
  card.atk += growthRates.atk;
  card.spd += growthRates.spd;
  card.level = nextLevel;

  savePlayerCard(playerId, card);
  return true;
}
```

```typescript
function fuseCard(mainCardId: string, subCardId: string): FusionResult | null {
  const mainCard = getPlayerCard(mainCardId);
  const subCard = getPlayerCard(subCardId);

  if (!mainCard || !subCard) return null;

  // 1. 验证主副卡品质相同
  if (mainCard.quality !== subCard.quality) return null;

  // 2. 计算并检查菌丝消耗
  const fusionCost = getFusionCost(mainCard.quality, mainCard.level);
  if (!hasEnoughMycelium(mainCard.ownerId, fusionCost)) return null;

  // 3. 扣除菌丝
  deductMycelium(mainCard.ownerId, fusionCost);

  // 4. 调用 FusionSystem 执行融合
  //    FusionSystem 内部处理：属性继承、技能融合、品质提升概率等
  const result = FusionSystem.fuse(mainCard, subCard);

  // 5. 返回融合结果
  return result;
}
```

### 6. 队伍编组逻辑（3+1）

```typescript
interface Team {
  teamId: string;
  memberSlots: Array<{
    slotIndex: number;      // 0-2 为融合姬槽位，3 为强化芯片槽位
    cardId?: string;        // 融合姬卡牌ID
    chipId?: string;        // 强化芯片ID（仅槽位3）
  }>;
}

// 编组规则
function setTeamMember(team: Team, slotIndex: number, cardId: string): boolean {
  // 槽位0-2：融合姬
  if (slotIndex < 3) {
    // 检查是否已编入其他队伍
    if (isCardInOtherTeam(cardId, team.teamId)) return false;

    team.memberSlots[slotIndex].cardId = cardId;
    team.memberSlots[slotIndex].chipId = undefined;
    return true;
  }

  // 槽位3：强化芯片
  if (slotIndex === 3) {
    team.memberSlots[slotIndex].chipId = cardId;
    team.memberSlots[slotIndex].cardId = undefined;
    return true;
  }

  return false;
}

// 队伍验证
function validateTeam(team: Team): boolean {
  const members = team.memberSlots.slice(0, 3).filter(s => s.cardId);
  const chip = team.memberSlots[3];

  // 至少1个融合姬
  if (members.length === 0) return false;

  // 强化芯片槽可选
  if (chip.chipId) {
    // 检查芯片是否存在
    if (!getChipConfig(chip.chipId)) return false;
  }

  return true;
}
```
